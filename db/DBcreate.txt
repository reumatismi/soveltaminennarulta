CREATE TABLE proj_user(
 id INT AUTO_INCREMENT,
 username VARCHAR(15) UNIQUE,
 firstname VARCHAR(50),
 lastname VARCHAR(50),
 role INT,
 classid VARCHAR(3),
 password VARCHAR(50),
 vst DATE,
 vet DATE NULL,
 PRIMARY KEY(id, vst)
)engine=myisam;

CREATE TABLE proj_role(
 id INT,
 description VARCHAR(20),
 PRIMARY KEY(id)
)engine=myisam;

CREATE TABLE proj_class(
 classid CHAR(3),
 class CHAR(2),
 vst DATE,
 vet DATE NULL,
 PRIMARY KEY(classid, vst)
)engine=myisam;

CREATE TABLE proj_mediafeed(
 id INT AUTO_INCREMENT,
 classid CHAR(3),
 userid INT,
 mediafilename TEXT,
 mediadesc CHAR(100),
 visibility INT,
 vst TIMESTAMP,
 vet TIMESTAMP NULL,
 PRIMARY KEY(id, vst)
)engine=myisam;

CREATE TABLE proj_comment(
 id INT AUTO_INCREMENT,
 userid INT,
 mediaid INT,
 commenttext Char(100),
 visibility INT,
 vst TIMESTAMP,
 vet TIMESTAMP NULL,
 PRIMARY KEY(id, vst)
)engine=myisam;

CREATE TABLE proj_games(
 gameid INT AUTO_INCREMENT,
 gamename CHAR(50),
 gametype INT,
 vst DATE,
 vet DATE NULL,
 PRIMARY KEY(gameid, vst)
)engine=myisam;

CREATE TABLE proj_gamedesc(
 typeid INT AUTO_INCREMENT,
 type CHAR(50),
 PRIMARY KEY(typeid)
)engine=myisam;

CREATE TABLE proj_gamestats(
 gameid INT,
 points INT,
 playerid INT,
 vst TIMESTAMP,
 vet TIMESTAMP,
 PRIMARY KEY(gameid, playerid, vst)
)engine=myisam;

CREATE TABLE proj_visibility(
 id INT,
 description CHAR(10),
 PRIMARY KEY(id)
)engine=myisam;
